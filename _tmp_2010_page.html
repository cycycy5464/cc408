  int j = to;

    a[i] = a[j];
    a[j] = tmp;
    i++;
    j--;
  }
}

  if (p < 0) {
    return;

  reverse(a, 0, p-1);
  reverse(a, p, n-1);
  reverse(a, 0, n-1);
